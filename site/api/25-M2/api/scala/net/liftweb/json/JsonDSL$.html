net.liftweb.json

JsonDSL

object JsonDSL extends JsonDSL

A DSL to produce valid JSON. Example: 

import net.liftweb.json.JsonDSL._
("name", "joe") ~ ("age", 15) == JObject(JField("name",JString("joe")) :: JField("age",JInt(15)) :: Nil)
